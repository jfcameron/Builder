cmake_minimum_required (VERSION 3.1)
cmake_policy(SET CMP0037 NEW)

include("lib/debug.cmake")
include("lib/json.cmake")
include("lib/filter.cmake")

# Check bash args, bail if required are not present
if (NOT DEFINED ${PLATFORM})
    debug_error("A platform must be selected!")
    #return()
endif()

# Parse Config.json
file(STRINGS ./config.json configFileString)
json_parse(Config configFileString)
debug_log("Config json:")
foreach(var ${Config})
    debug_log("${var} = ${${var}}")
endforeach()

#Set consts
set(SOURCE_ROOT "../src")

#Collect, sort and print common files
file(GLOB_RECURSE COMMON_SOURCE_FILES
    "${SOURCE_ROOT}/common/*.cpp"
    "${SOURCE_ROOT}/common/*.h"
)
debug_log("Common files:")
foreach(file_path ${COMMON_SOURCE_FILES})
    debug_log(${file_path})
endforeach()
filter_assignSourceGroups(${COMMON_SOURCE_FILES})

#Collect, sort and print platform files
file(GLOB_RECURSE PLATFORM_SPECIFIC_SOURCE_FILES
    "${SOURCE_ROOT}/${PLATFORM}/*.cpp"
    "${SOURCE_ROOT}/${PLATFORM}/*.h"
)
debug_log("${PLATFORM} specific files:")
foreach(file_path ${PLATFORM_SPECIFIC_SOURCE_FILES})
    debug_log(${file_path})
endforeach()
filter_assignSourceGroups(${PLATFORM_SPECIFIC_SOURCE_FILES})

#Build the project
project ("${Config.ProjectName}")
add_executable("${Config.ProjectName}" ${COMMON_SOURCE_FILES} ${PLATFORM_SPECIFIC_SOURCE_FILES})
